function void build_phase(uvm_phase phase);
if(!uvm_config_db #(env_config)::get(this,"","env_config",m_cfg))
`uvm_fatal("sb","cannot get env config, have you set it?");

axi_agnth=new[m_cfg.no_of_agents];
slv_agenth=new[m_cfg.no_of_slave_agents];

foreach(axi_agnth[i])
	axi_agnth[i]=new($sformatf("axi_agnth[%0d]",i),this);

foreach(slv_agenth[i])
	slv_agenth[i]=new($sformatf("slv_agenth[%0d]",i),this);
super.build_phase(phase);
endfunction

task run_phase(uvm_phase phase);
forever
	begin
		fork
			begin
				fork:A
					begin
						axi_agnth[0].get(wr_xtn);
						q1.push_back(wr_xtn);
						write_axi_cg.sample();
						foreach(wr_xtn.WDATA[i])
							begin
								write_axi_cg1.sample(i);
							end
					end
					begin
						axi_agnth[1].get(wr_xtn);
						q1.push_back(wr_xtn);
						write_axi_cg.sample();
						foreach(wr_xtn.WDATA[i])
							begin
								write_axi_cg1.sample(i);
							end
																	end
					begin
						axi_agnth[2].get(wr_xtn);
						q1.push_back(wr_xtn);
						write_axi_cg.sample();
						foreach(wr_xtn.WDATA[i])
							begin
								write_axi_cg1.sample(i);
							end
					end
					begin
						axi_agnth[3].get(wr_xtn);
						q1.push_back(wr_xtn);
						write_axi_cg.sample();
						foreach(wr_xtn.WDATA[i])
							begin
								write_axi_cg1.sample(i);
							end
					end
				join_any
				disable A;

				fork:B
					begin
						slv_agenth[0].get(rd_xtn);
						q1.push_back(rd_xtn);
						read_axi_cg.sample();
						foreach(rd_xtn.WDATA[i])
							begin
								read_axi_cg1.sample(i);
							end
					end
					begin
						slv_agenth[1].get(rd_xtn);
						q1.push_back(rd_xtn);
						read_axi_cg.sample();
						foreach(rd_xtn.WDATA[i])
							begin
								read_axi_cg1.sample(i);
							end
					end
					begin
						slv_agenth[2].get(rd_xtn);
						q1.push_back(rd_xtn);
						read_axi_cg.sample();
						foreach(rd_xtn.WDATA[i])
							begin
								read_axi_cg1.sample(i);
							end
					end
					begin
						slv_agenth[3].get(rd_xtn);
						q1.push_back(rd_xtn);
						read_axi_cg.sample();
						foreach(rd_xtn.WDATA[i])
							begin
								read_axi_cg1.sample(i);
							end
					end
					join_any
					disable B;
				end
				join

	pkt_rcvd++;
				check_data(wr_xtn,rd_xtn);
				end
endtask

task check_data(axi_xtn mst_xtn, axi_xtn slv_xtn);
	if(mst_xtn.compare(slv_xtn))
		begin
			`uvm_info("SCOREBOARD22","data compare successfull",UVM_LOW)
		pkt_cmprd++;
		end
	else
			`uvm_info("SCOREBOARD11","data compare unsuccessfull",UVM_LOW)

endtask

function void report_phase(uvm_phase phase);
	`uvm_info("SCOREBOARD",$sformatf("No.of packets received:%0d",pkt_rcvd),UVM_LOW);

	`uvm_info("SCOREBOARD",$sformatf("No.of packets received:%0d",pkt_cmprd),UVM_LOW);
endfunction





	

endclass
